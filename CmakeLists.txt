cmake_minimum_required(VERSION 3.20)

project(RayTracer)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include_directories(
    include
    third_party
)

find_package(SDL2 REQUIRED)

add_executable(RayTracer
    src/main.cpp
    src/vec3.cpp
    src/ray.cpp
    src/renderer.cpp
)

target_link_libraries(RayTracer PRIVATE SDL2::SDL2)
